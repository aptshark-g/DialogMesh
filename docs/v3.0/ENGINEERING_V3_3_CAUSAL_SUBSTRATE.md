能量/信息/资源的源头\n    SINK = "sink"               # 能量/信息/资源的终点\n    STORE_POTENTIAL = "store:potential"  # 势变量存储\n    STORE_KINETIC = "store:kinetic"     # 流变量存储\n    DISSIPATE = "dissipate"     # 能量耗散/信息损失\n    TRANSFORM = "transform"     # 能量/信息形式转换\n    JUNCTION_SUM = "junction_sum"   # 多输入 -> 一输出\n    JUNCTION_SPLIT = "junction_split" # 一输入 -> 多输出\n\n@dataclass\nclass CausalConstraints:\n    """从编译器输出中提取的约束特征"""\n    domain_hint: str = "general"\n    has_feedback: bool = False\n    involves_dissipation: bool = False\n    involves_storage: bool = False\n    causal_direction: str = "cause->effect"\n    involves_transformation: bool = False\n\n@dataclass\nclass SkeletonMatch:\n    """骨架匹配结果"""\n    roles: list[MetaRole]          # 骨架中包含的角色列表\n    coverage: float                # 约束满足度 [0,1]\n    score: float                   # 综合匹配度 [0,1]\n    is_multi_match: bool = False   # 多个候选得分接近\n\n    def to_structural_prior(self) -> float:\n        if self.score > 0.8: return 0.7\n        if self.score > 0.5: return 0.3\n        return 0.0\n```\n\n---\n\n## 4. MetaRoles Enum + Domain Mappings (meta_roles.py)\n\n### 4.1 职责\n\n定义 8 个元角色枚举 + 跨领域映射表。\n\n```python\nclass MetaRoles:\n    """8 元角色定义 + 跨领域映射"""\n\n    ALL = [\n        MetaRole.SOURCE, MetaRole.SINK,\n        MetaRole.STORE_POTENTIAL, MetaRole.STORE_KINETIC,\n        MetaRole.DISSIPATE, MetaRole.TRANSFORM,\n        MetaRole.JUNCTION_SUM, MetaRole.JUNCTION_SPLIT,\n    ]\n\n    DOMAIN_MAP = {  # domain -> applicable roles\n        "physical": [MetaRole.SOURCE, MetaRole.SINK, MetaRole.DISSIPATE,\n                      MetaRole.STORE_POTENTIAL, MetaRole.STORE_KINETIC,\n                      MetaRole.TRANSFORM, MetaRole.JUNCTION_SUM, MetaRole.JUNCTION_SPLIT],\n        "software": [MetaRole.SOURCE, MetaRole.SINK, MetaRole.STORE_POTENTIAL,\n                      MetaRole.STORE_KINETIC, MetaRole.DISSIPATE, MetaRole.TRANSFORM,\n                      MetaRole.JUNCTION_SUM, MetaRole.JUNCTION_SPLIT],\n        "organizational": [MetaRole.SOURCE, MetaRole.SINK, MetaRole.STORE_POTENTIAL,\n                           MetaRole.DISSIPATE, MetaRole.TRANSFORM,\n                           MetaRole.JUNCTION_SUM, MetaRole.JUNCTION_SPLIT],\n        "general": ALL,\n    }\n\n    DESCRIPTIONS = {  # 中文描述\n        MetaRole.SOURCE: "能量/信息/资源的源头",\n        MetaRole.SINK: "能量/信息/资源的终点",\n        MetaRole.STORE_POTENTIAL: "势变量存储 (电容/缓存)",\n        MetaRole.STORE_KINETIC: "流变量存储 (电感/队列)",\n        MetaRole.DISSIPATE: "能量耗散/信息损失",\n        MetaRole.TRANSFORM: "能量/信息形式转换",\n        MetaRole.JUNCTION_SUM: "多输入 -> 一输出汇聚",\n        MetaRole.JUNCTION_SPLIT: "一输入 -> 多输出分发",\n    }\n\n    @classmethod\n    def get_domain_roles(cls, domain: str) -> list[MetaRole]:\n        return cls.DOMAIN_MAP.get(domain, cls.DOMAIN_MAP["general"])\n```\n\n---\n\n## 5. SkeletonLibrary\n\n### 5.1 职责\n\n预置 ~20 个常见因果骨架, 每个骨架是元角色的有向组合。\n\n### 5.2 接口\n\n```python\n@dataclass\nclass CausalSkeleton:\n    name: str\n    roles: list[MetaRole]\n    requires: list[str]  # 约束要求列表\n    description: str\n\nclass SkeletonLibrary:\n    """因果骨架库 (~20 个常见模式)"""\n\n    def __init__(self):\n        self.skeletons: list[CausalSkeleton] = []\n        self._load_default()\n\n    def _load_default(self):\n        self.skeletons = [\n            CausalSkeleton(\n                "source_dissipate",\n                [MetaRole.SOURCE, MetaRole.DISSIPATE, MetaRole.SINK],\n                ["involves_dissipation"],\n                "能量从源头经耗散到终点"\n            CausalSkeleton(\n                "source_store_sink",\n                [MetaRole.SOURCE, MetaRole.STORE_POTENTIAL, MetaRole.SINK],\n                ["involves_storage"],\n                "能量从源头经存储到终点"\n            CausalSkeleton(\n                "source_transform_sink",\n                [MetaRole.SOURCE, MetaRole.TRANSFORM, MetaRole.SINK],\n                ["involves_transformation"],\n                "能量从源头经转换到终点"\n            CausalSkeleton(\n                "feedback_loop",\n                [MetaRole.SOURCE, MetaRole.TRANSFORM, MetaRole.SINK, MetaRole.SOURCE],\n                ["has_feedback"],\n                "反馈环: 输出回馈到输入"\n            CausalSkeleton(\n                "parallel_processing",\n                [MetaRole.SOURCE, MetaRole.JUNCTION_SPLIT, MetaRole.TRANSFORM, MetaRole.JUNCTION_SUM, MetaRole.SINK],\n                [],\n                "并行处理: 分 -> 并行 -> 合"\n            CausalSkeleton(\n                "buffered_flow",\n                [MetaRole.SOURCE, MetaRole.STORE_KINETIC, MetaRole.TRANSFORM, MetaRole.SINK],\n                ["involves_storage"],\n                "带缓冲的流转: 源 -> 缓冲 -> 转换 -> 终点"\n            CausalSkeleton(\n                "cascade_transform",\n                [MetaRole.SOURCE, MetaRole.TRANSFORM, MetaRole.TRANSFORM, MetaRole.SINK],\n                ["involves_transformation"],\n                "级联转换: 多步转换链"\n        ]\n        # 后续可扩展至 ~20 个\n\n    def query(self, constraints: CausalConstraints) -> list[CausalSkeleton]:\n        candidates = []\n        for sk in self.skeletons:\n            matches = sum(1 for r in sk.requires if getattr(constraints, r, False))\n            total = len(sk.requires) if sk.requires else 1\n            coverage = matches / total\n            if coverage > 0:  # 至少匹配一条约束\n                candidates.append((sk, coverage))\n        candidates.sort(key=lambda x: -x[1])\n        return [c[0] for c in candidates[:5]]\n```\n\n---\n\n## 6. SkeletonMatcher (Step 2 + Step 3)\n\n### 6.1 职责\n\nStep2: 约束条件 -> 候选骨架 (SkeletonLibrary.query)\nStep3: 匹配度排序 -> structural_prior\n\n### 6.2 接口\n\n```python\nclass ConstraintExtractor:\n    """Step 1: 从编译器输出中提取约束特征"""\n\n    def extract(self, compiler_output: ParseResult) -> CausalConstraints:\n        if compiler_output.undefined or not compiler_output.is_reliable:\n            return CausalConstraints()  # 空约束 -> structural_prior = 0\n\n        slots = compiler_output.slots\n        constraints = CausalConstraints()\n\n        # 从 slots 中推断约束\n        action = slots.get("action")\n        if action:\n            action_map = {\n                "运行": ("software", False, True, False, "cause->effect", True),\n                "修改": ("software", False, False, False, "cause->effect", True),\n                "分析": ("software", True, False, True, "bidirectional", False),\n                "查看": ("software", False, False, False, "cause->effect", False),\n                "执行": ("software", False, True, False, "cause->effect", True),\n                "创建": ("software", False, False, True, "cause->effect", True),\n            }\n            mapped = action_map.get(action.value)\n            if mapped:\n                constraints.domain_hint = mapped[0]\n                constraints.has_feedback = mapped[1]\n                constraints.involves_dissipation = mapped[2]\n                constraints.involves_storage = mapped[3]\n                constraints.causal_direction = mapped[4]\n                constraints.involves_transformation = mapped[5]\n\n        return constraints\n\nclass SkeletonMatcher:\n    """Step 2 + Step 3: 约束 -> 候选 -> 匹配度 -> prior"""\n\n    def __init__(self, library: SkeletonLibrary):\n        self.library = library\n\n    def match(\n        self, constraints: CausalConstraints\n    ) -> SkeletonMatch | None:\n        if not constraints.domain_hint:\n            return None\n\n        # Step 2: 查询候选骨架\n        candidates = self.library.query(constraints)\n        if not candidates:\n            return None\n\n        # 计算匹配度\n        scored = []\n        for sk in candidates:\n            matched = sum(1 for r in sk.requires if getattr(constraints, r, False))\n            total = len(sk.requires) if sk.requires else 1\n            coverage = matched / total\n            scored.append((sk, coverage))\n\n        scored.sort(key=lambda x: -x[1])\n\n        # Step 3: 选择最佳匹配\n        best_sk, best_coverage = scored[0]\n\n        # 检查是否有多候选得分接近\n        is_multi = False\n        if len(scored) > 1:\n            second_coverage = scored[1][1]\n            if best_coverage - second_coverage < 0.15:\n                is_multi = True\n\n        # score 在 coverage 基础上做一些平滑\n        score = best_coverage\n\n        return SkeletonMatch(\n            roles=best_sk.roles, coverage=best_coverage,\n            score=score, is_multi_match=is_multi\n        )\n\n    def match_and_get_prior(\n        self, compiler_output: ParseResult\n    ) -> tuple[float, SkeletonMatch | None]:\n        extractor = ConstraintExtractor()\n        constraints = extractor.extract(compiler_output)\n        if not constraints.domain_hint:\n            return (0.0, None)\n        match = self.match(constraints)\n        if not match or match.score <= 0.5:\n            return (0.0, match)\n        return (match.to_structural_prior(), match)\n```\n\n---\n\n## 7. DeltaAdjuster\n\n### 7.1 职责\n\nReflective 周期检查: 根据 structural_prior 和 correction_count 动态调整 δ。\n\n### 7.2 接口\n\n```python\nclass DeltaAdjuster:\n    INITIAL_DELTA = 0.05\n    MAX_DELTA = 0.15\n    MIN_DELTA = 0.0\n\n    def __init__(self):\n        self.current_delta = self.INITIAL_DELTA\n\n    def adjust(\n        self, edge, reflective_cycle: int\n    ) -> float:\n        if reflective_cycle % 50 != 0:\n            return self.current_delta  # 每 50 轮检查一次\n\n        if edge.structural_prior > 0.3 and edge.correction_count < 2:\n            # 高先验 + 低纠正 -> 上调\n            self.current_delta = min(self.MAX_DELTA, self.current_delta + 0.02)\n        elif edge.correction_count > 5:\n            # 高纠正 -> 下调\n            self.current_delta = max(self.MIN_DELTA, self.current_delta - 0.02)\n\n        return self.current_delta\n\n    def get_status(self) -> dict:\n        return {"current_delta": self.current_delta}\n```\n\n---\n\n## 8. CausalSubstrate (入口)\n\n### 8.1 职责\n\n检查触发条件（行为链 > 10 步）, 执行三步映射, 写入 structural_prior 到 BehaviorEdge。\n\n### 8.2 接口\n\n```python\nclass CausalSubstrate:\n    """因果基地元角色入口"""\n\n    MIN_CHAIN_LENGTH = 10\n    MIN_CHAIN_FOR_OFFLINE = 5\n\n    def __init__(self, graph: BehaviorGraph, library=None, adjuster=None):\n        self.graph = graph\n        self.library = library or SkeletonLibrary()\n        self.matcher = SkeletonMatcher(self.library)\n        self.adjuster = adjuster or DeltaAdjuster()\n\n    def should_trigger(self, chain_length: int) -> bool:\n        """行为链长度 > 10 -> 触发离线映射"""\n        return chain_length > self.MIN_CHAIN_LENGTH\n\n    def process_chain(\n        self, chain: list[BehaviorStep],\n        compiler_outputs: list[ParseResult],\n        reflective_cycle: int\n    ) -> dict[str, float]:\n        """处理一条行为链, 返回 edge_key -> structural_prior 映射"""\n        if not self.should_trigger(len(chain)):\n            return {}\n\n        results = {}\n        for step, compiler_out in zip(chain, compiler_outputs):\n            prior, match = self.matcher.match_and_get_prior(compiler_out)\n            if prior > 0:\n                # 写入 BehaviorEdge\n                for edge_key, edge in self.graph.edges.items():\n                    to_step = self.graph.nodes.get(edge.to_step_id)\n                    if to_step and to_step.step_id == step.step_id:\n                        edge.structural_prior = prior\n                        delta = self.adjuster.adjust(edge, reflective_cycle)\n                        results[edge_key] = prior\n                        if match and match.is_multi_match:\n                            # 多个候选接近 -> 降低 δ\n                            self.adjuster.current_delta = max(\n                                DeltaAdjuster.MIN_DELTA,\n                                self.adjuster.current_delta - 0.01\n                            )\n        return results\n\n    def process_single(\n        self, step: BehaviorStep,\n        compiler_out: ParseResult\n    ) -> float:\n        """处理单步 (在线调用)"""\n        prior, _ = self.matcher.match_and_get_prior(compiler_out)\n        return prior\n```\n\n---\n\n##
